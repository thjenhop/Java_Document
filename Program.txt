
#include <iostream>
#include <vector>

using namespace std;

class Solustion
{
public:
    int searchInsert(vector<int> &nums, int taget)
    {
        int i;
        for(i = 0; i < nums.size(); i++)
        {
            if(nums.at(i) == taget)
            {
                return i;
            }
            if(nums.at(i) > taget)
            {
                return i;
            }
        }
        return i;
    }
};

int main()
{
    Solustion x;
    vector<int> v = {1, 3, 5, 6};
    int n;
    cin >> n;
    cout << x.searchInsert(v, n);
    return 0;
}
